354template<class T>

356 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++)
357 if ((*it)->info == in) {
358 auto v = *it;
359 vertexSet.erase(it);
360 for (auto u: vertexSet)
361 u->removeEdgeTo(v);
362 delete v;
363 return true;

392 v->visited = true;

394 for (auto &e: v->adj) {

423/****************** BFS ********************/
424/*
425 * Performs a breadth-first search (bfs) in a graph (this), starting
426 * from the vertex with the given source contents (source).
427 * Returns a vector with the contents of the vertices by bfs order.
428 */
429template<class T>
430vector<T> Graph<T>::bfs(const T &source) const {
431 vector<T> res;
432 auto s = findVertex(source);
433 if (s == NULL)
434 return res;
435 queue<Vertex<T> *> q;
436 for (auto v: vertexSet)
437 v->visited = false;
438 q.push(s);
439 s->visited = true;
440 while (!q.empty()) {
441 auto v = q.front();
442 q.pop();
443 res.push_back(v->info);
444 for (auto &e: v->adj) {
445 auto w = e.dest;
446 if (!w->visited) {
447 q.push(w);
448 w->visited = true;
449 }
450 }
451 }
452 return res;
453}

456/****************** isDAG ********************/
457/*
458 * Performs a depth-first search in a graph (this), to determine if the graph
459 * is acyclic (acyclic directed graph or DAG).
460 * During the search, a cycle is found if an edge connects to a vertex
461 * that is being processed in the stack of recursive calls (see theoretical classes).
462 * Returns true if the graph is acyclic, and false otherwise.
463 */
464
465template<class T>
466bool Graph<T>::isDAG() const {
467 for (auto v: vertexSet) {
468 v->visited = false;
469 v->processing = false;
470 }
471 for (auto v: vertexSet)
472 if (!v->visited)
473 if (!dfsIsDAG(v))
474 return false;
475 return true;
476}

482template<class T>
483bool Graph<T>::dfsIsDAG(Vertex<T> *v) const {
484 v->visited = true;
485 v->processing = true;
486 for (auto &e: v->adj) {
487 auto w = e.dest;
488 if (w->processing)
489 return false;
490 if (!w->visited)
491 if (!dfsIsDAG(w))
492 return false;
493 }
494 v->processing = false;
495 return true;
