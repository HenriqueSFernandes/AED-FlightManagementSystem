473/****************** BFS ********************/
474/*
475 * Performs a breadth-first search (bfs) in a graph (this), starting
476 * from the vertex with the given source contents (source).
477 * Returns a vector with the contents of the vertices by bfs order.
478 */
479template<class T>
480vector<T> Graph<T>::bfs(const T &source) const {
481 vector<T> res;
482 auto s = findVertex(source);
483 if (s == NULL)
484 return res;
485 queue<Vertex<T> *> q;
486 for (auto v: vertexSet)
487 v->visited = false;
488 q.push(s);
489 s->visited = true;
490 while (!q.empty()) {
491 auto v = q.front();
492 q.pop();
493 res.push_back(v->info);
494 for (auto &e: v->adj) {
495 auto w = e.dest;
496 if (!w->visited) {
497 q.push(w);
498 w->visited = true;
499 }
500 }
501 }
502 return res;
503}

506vector<T> Graph<T>::bfsLimited(const T &source, int k) const {
507 vector<T> res;
508 auto s = findVertex(source);
509 if (s == NULL)
510 return res;
511 queue<Vertex<T> *> q;
512 for (auto v: vertexSet) {
513 v.second->visited = false;
514 v.second->setNum(0);
515
516 }
517 q.push(s);
518 s->visited = true;
519 s->setNum(0);
520 while (!q.empty()) {
521 auto v = q.front();
522 q.pop();
523 for (auto &e: v->adj) {
524 auto w = e.dest;
525 if (!w->visited) {
526 q.push(w);
527 w->visited = true;
528 w->setNum(v->getNum() + 1);
529 if (w->getNum() <= k) {
530 res.push_back(w->info);
531 }
532 }
533 }
534 }

539/****************** isDAG ********************/
540/*
541 * Performs a depth-first search in a graph (this), to determine if the graph
542 * is acyclic (acyclic directed graph or DAG).
543 * During the search, a cycle is found if an edge connects to a vertex
544 * that is being processed in the stack of recursive calls (see theoretical classes).
545 * Returns true if the graph is acyclic, and false otherwise.
546 */
547
548template<class T>
549bool Graph<T>::isDAG() const {
550 for (auto v: vertexSet) {
551 v->visited = false;
552 v->processing = false;
553 }
554 for (auto v: vertexSet)
555 if (!v->visited)
556 if (!dfsIsDAG(v))
557 return false;
558 return true;
559}

565template<class T>
566bool Graph<T>::dfsIsDAG(Vertex<T> *v) const {
567 v->visited = true;
568 v->processing = true;
569 for (auto &e: v->adj) {
570 auto w = e.dest;
571 if (w->processing)
572 return false;
573 if (!w->visited)
574 if (!dfsIsDAG(w))
575 return false;
576 }
577 v->processing = false;
578 return true;
579}
